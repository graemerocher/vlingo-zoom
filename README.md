The vlingo/PLATFORM adoption accelerator for Java and other JVM languages.

### Bintray

```xml
  <repositories>
    <repository>
      <id>jcenter</id>
      <url>https://jcenter.bintray.com/</url>

  </repositories>

    <dependency>
      <groupId>io.vlingo</groupId>

    </dependency>
  </dependencies>
```

```gradle
dependencies {
```
